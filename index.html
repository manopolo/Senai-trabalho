-- Criar a Base de Dados CREATE DATABASE saep_saude; USE saep_saude; -- Tabela de Utilizadores (Clientes e Funcionários) CREATE TABLE usuarios ( id INT AUTO_INCREMENT PRIMARY KEY, nome VARCHAR(100) NOT NULL, email VARCHAR(100) UNIQUE NOT NULL, senha VARCHAR(255) NOT NULL, foto_perfil VARCHAR(255) ); -- Tabela de Atividades CREATE TABLE atividades ( id INT AUTO_INCREMENT PRIMARY KEY, usuario_id INT NOT NULL, tipo ENUM('corrida', 'caminhada', 'trilha') NOT NULL, -- Aceita apenas estes valores conforme a regra 3.1 distancia_metros INT NOT NULL, -- Valores em metros duracao_minutos INT NOT NULL, -- Valores em minutos calorias INT NOT NULL, data_criacao DATETIME DEFAULT CURRENT_TIMESTAMP, curtidas INT DEFAULT 0, comentarios INT DEFAULT 0, FOREIGN KEY (usuario_id) REFERENCES usuarios(id) );
