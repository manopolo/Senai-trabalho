if (campo.value.trim() === '') { campo.classList.add('erro-input'); erro.style.display = 'block'; // Mostra "email ou senha obrigatório" valido = false; } else { campo.classList.remove('erro-input'); erro.style.display = 'none'; } }); if (valido) { // Aqui entraria a chamada fetch() para o backend alert("Login efetuado (Simulação)"); fecharModal('modalLogin'); } } // --- LÓGICA DE COMENTÁRIOS E LIKES --- function darLike(botao) { // Muda a cor do like para vermelho e aumenta o número (Regras 3.1 a 3.4) botao.classList.toggle('like-ativo');
